let flatten a = 
   let len = ref 0 in
   for i = 0 to length a - 1 do
      len := !len + length (unsafe_get a i)
   done;
   let i,j = ref 0, ref 0 in
   init !len 
   (fun _ ->
      while !j = length (unsafe_get a !i) do
         i := !i + 1;
         j := 0
      done;
      let v = unsafe_get (unsafe_get a !i) !j in
      j := !j + 1;
      v)
