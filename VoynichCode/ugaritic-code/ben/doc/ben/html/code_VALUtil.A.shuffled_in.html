let shuffled_in ?(seed=None) a =
   if length a = 0 then 
      raise (Invalid_argument "can't shuffle an empty array");
   (match seed with 
      | Some n -> Random.init n
      | _ -> ());
   let rec fy a = function  (* fisher-yates shuffle algorithm *)
      | 0 -> a
      | i -> 
         let swap_pos = Random.int i in
         let temp = a.(i) in
         a.(i) <- a.(swap_pos);
         a.(swap_pos) <- temp;
         fy a (pred i) 
   in
   fy a (length a - 1)
