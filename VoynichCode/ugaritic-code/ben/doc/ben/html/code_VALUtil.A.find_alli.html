let find_alli p a =
   let n = length a in
   let bs = BitSet.create n in
   for i = 0 to n-1 do
      if p a.(i) then BitSet.set bs i
   done;
   let n' = BitSet.count bs in
   let j = ref 0 in
   let a' = init n'
      (fun _ ->
         while not (BitSet.is_set bs !j) do incr j done;
         let r = !j in
         incr j; 
         r) in
   a'
