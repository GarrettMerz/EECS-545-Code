let lines_of_str ?(comment_str="#") str = 
   S.strip ~chars:"\n" str                                        |>
   (fun s -> S.nsplita s "\n")                                    |>
   A.filter (fun s-> not (S.starts_with s comment_str))           |>
   A.map (fun s -> try fst (S.split s comment_str) with _ -> s)
