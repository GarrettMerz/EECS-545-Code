let of_file key_fun val_fun fn =
   let lines = Std.input_list (open_in fn) in
   let res = create 500 in
   L.iter 
      (fun l ->
         try 
            let key_s, val_s = String.split l "\t" in
            add res (key_fun key_s) (val_fun val_s)
         with _ -> 
            raise (Invalid_argument "each line in file must have a SINGLE TAB character separating key from value"))
      lines;
   res
